203 // Purpose : Execute the top level DC sweep control loop.
204 // Special Notes :
205 // Scope : public
206 // Creator : Todd S. Coffey, 9214

208 //-----------------------------------------------------------------------------
209 bool N_TIA_MPDEInterface::runDCOP()
210 {
211  if( !tiaParams_.NOOP )

214  // try and do the operating point calculation
215  anaManagerPtr_->analysis = Xyce::Analysis::ANP_MODE_DC_SWEEP;
216  return anaManagerPtr_->run();
217  }
218  return false;
219 }
