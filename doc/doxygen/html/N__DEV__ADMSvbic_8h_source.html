399 class AnalogFunctions
400 {
401 public:
402 
403  // Analog Function limRTH
404  template<typename ScalarT> ScalarT limRTH(ScalarT orig, ScalarT old)
405  {
406 
407 
408  ScalarT limRTH;
409  ScalarT t0;
410  ScalarT t1;
411  ScalarT retval;
412  {
413  t0 = (orig-old);
414  t1 = fabs(t0);
415  retval = orig;
416  if ((t1>5.0))
417  {
418  if ((t0>0))
419  {
420  retval = (old+5.0);
421  }
422  else
423  {
424  retval = (old-5.0);
425  }
426  }
427  limRTH = retval;
428  }
429  return(limRTH);
430  }
431 
432 };
