145  //---------------------------------------------------------------------------
146  // Function : areStateVectors
147  // Purpose : To compute a Jacobian, the state vectors must be
148  // updated with respect to the solution in the group.
149  // However, the state vectors are updated ONLY during
150  // calls to compute the residual. This method checks
151  // to see if the state vectors still correspond to this
152  // group. Returns true if state vectors are correct.
153  //---------------------------------------------------------------------------
154  inline bool areStateVectors(const Group* grp) const
155  {
156  return (grp == ownerOfStateVectors_);
157  };
